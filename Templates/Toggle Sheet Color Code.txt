        Dim Doc As DrawingDocument = g_inventorApplication.ActiveDocument

        Dim TObj As TransientObjects = g_inventorApplication.TransientObjects

        'get the color of the sheet
        Dim Color As Color
        Color = Doc.SheetSettings.SheetColor

        'get the RGB from the color
        Dim RGB As String
        RGB = Color.Red & ", " & Color.Green & ", " & Color.Blue

        'toggle sheet color
        'if white, set to inventor default color
        If RGB = "255, 255, 255" Then
            Color = TObj.CreateColor(237, 237, 214)
            'if inventor default color, set to white
        ElseIf RGB = "237, 237, 214" Then
            Color = TObj.CreateColor(255, 255, 255)
        Else
            'if something else set to inventor default color
            Color = TObj.CreateColor(237, 237, 214)
        End If

        Doc.SheetSettings.SheetColor = Color
